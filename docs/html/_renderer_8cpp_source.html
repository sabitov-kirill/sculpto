27void scl::renderer::DrawDepth(const shared<mesh> &Mesh, const matr4 &Transform)
28{
29 if (!Mesh->IsCastingShadow) return;
30
31 Pipeline.ShadowPassShader->Bind();
32 for (auto &submesh : Mesh->SubMeshes)
33 {
34 // Currently local tranform matrix is usless.
35 // matr4 world = submesh.LocalTransform * Transform;
36
37 Pipeline.ShadowPassShader->SetMatr4("u_MatrWVP", Transform * matr4(Pipeline.LightsStorage.DirectionalLight.ViewProjection));
38 render_bridge::DrawIndices(submesh.VertexArray);
39 }
40}

42void scl::renderer::DrawGeometry(const shared<mesh> &Mesh, const matr4 &Transform)
43{
44 if (!Mesh->IsDrawing) return;
45
46 for (auto &submesh : Mesh->SubMeshes)
47 {
48 // matr4 world = submesh.LocalTransform * Transform;
49 submesh.Material->Bind();
50 submesh.Material->Shader->SetMatr3("u_MatrN", matr3(Transform.Inverse().Transpose()));
51 submesh.Material->Shader->SetMatr4("u_MatrW", Transform);
52 submesh.Material->Shader->SetMatr4("u_MatrWVP", Transform * Pipeline.ViewProjection);
53 render_bridge::DrawIndices(submesh.VertexArray);
54 }
55}

78void scl::renderer::ApplyBluredTexture(const shared<frame_buffer> &Destination, const shared<texture_2d> &Source, int Iterations)
79{
80 Pipeline.GaussianBlurApplyShader->Bind();
81 for (int i = 0; i < Iterations; i++)
82 {
83 if (i % 2 == 0) Pipeline.BlurFrameBuffers0->Bind();
84 else Pipeline.BlurFrameBuffers1->Bind();
85
86 if (i == 0) Source->Bind(render_context::TEXTURE_SLOT_BLUR_BUFFER);
87 else if (i % 2 == 0) Pipeline.BlurFrameBuffers1->GetColorAttachment()->Bind(render_context::TEXTURE_SLOT_BLUR_BUFFER);
88 else Pipeline.BlurFrameBuffers0->GetColorAttachment()->Bind(render_context::TEXTURE_SLOT_BLUR_BUFFER);
89
90 Pipeline.GaussianBlurApplyShader->SetBool("u_IsHorizontal", i % 2);
91 DrawFullscreenQuad();
92
93 if (i % 2 == 0) Pipeline.BlurFrameBuffers0->Unbind();
94 else Pipeline.BlurFrameBuffers1->Unbind();
95 }
96
97 ApplyTexture(Destination, Iterations % 2 == 0 ? Pipeline.BlurFrameBuffers0->GetColorAttachment() : Pipeline.BlurFrameBuffers1->GetColorAttachment());
98}

120void scl::renderer::ComputateLighting()
121{
122 if (Pipeline.Data.IsHDR) Pipeline.HDRFrameBuffer->Clear(), Pipeline.HDRFrameBuffer->Bind();
123 else Pipeline.MainFrameBuffer->Clear(), Pipeline.MainFrameBuffer->Bind();
124
125 Pipeline.PhongLightingApplyShader->Bind();
126 Pipeline.GBuffer->GetColorAttachment(render_context::COLOR_ATTACHMENT_GEOM_PASS_OUT_POSITION )->Bind(render_context::TEXTURE_SLOT_GEOM_PASS_OUT_POSITION );
127 Pipeline.GBuffer->GetColorAttachment(render_context::COLOR_ATTACHMENT_GEOM_PASS_OUT_NORMAL )->Bind(render_context::TEXTURE_SLOT_GEOM_PASS_OUT_NORMAL );
128 Pipeline.GBuffer->GetColorAttachment(render_context::COLOR_ATTACHMENT_GEOM_PASS_OUT_COLOR )->Bind(render_context::TEXTURE_SLOT_GEOM_PASS_OUT_COLOR );
129 Pipeline.GBuffer->GetColorAttachment(render_context::COLOR_ATTACHMENT_GEOM_PASS_OUT_PHONG_DIFFUSE )->Bind(render_context::TEXTURE_SLOT_GEOM_PASS_OUT_PHONG_DIFFUSE );
130 Pipeline.GBuffer->GetColorAttachment(render_context::COLOR_ATTACHMENT_GEOM_PASS_OUT_PHONG_SPECULAR )->Bind(render_context::TEXTURE_SLOT_GEOM_PASS_OUT_PHONG_SPECULAR );
131 Pipeline.GBuffer->GetColorAttachment(render_context::COLOR_ATTACHMENT_GEOM_PASS_OUT_PHONG_SHININESS)->Bind(render_context::TEXTURE_SLOT_GEOM_PASS_OUT_PHONG_SHININESS);
132
133 Pipeline.DataBuffer->Update(&Pipeline.Data, sizeof(pipeline_data));
134 Pipeline.DataBuffer->Bind(render_context::BINDING_POINT_SCENE_DATA);
135
136 Pipeline.LightsStorageBuffer->Update(&Pipeline.LightsStorage, sizeof(lights_storage));
137 Pipeline.LightsStorageBuffer->Bind(render_context::BINDING_POINT_LIGHTS_STORAGE);
138
139 if (Pipeline.LightsStorage.IsDirectionalLight &&
140 Pipeline.LightsStorage.DirectionalLight.IsShadows)
141 Pipeline.ShadowMap->GetDepthAttachment()->Bind(render_context::TEXTURE_SLOT_SHADOW_MAP);
142
143 DrawFullscreenQuad();
144
145 if (Pipeline.Data.IsHDR) Pipeline.HDRFrameBuffer->Unbind();
146 else Pipeline.MainFrameBuffer->Unbind();
147}

174void scl::renderer::StartPipeline(const camera &Camera, const vec3 &EnviromentAmbient)
175{
176 if (!Pipeline.IsInitialized) Pipeline.Initalize();
177
178 Pipeline.Data.Time = timer::GetTime();

180
181 Pipeline.Data.ViewportWidth = Camera.GetViewportWidth();
182 Pipeline.Data.ViewportHeight = Camera.GetViewportHeight();
183 Pipeline.Data.CameraPosition = Camera.GetPosition();
184 Pipeline.Data.CameraDirection = Camera.GetDirection();
185 Pipeline.ViewProjection = Camera.GetViewProjection();
186 Pipeline.Data.Exposure = Camera.Effects.Exposure;
187 Pipeline.Data.IsHDR = Camera.Effects.HDR;
188 Pipeline.Data.IsBloom = Camera.Effects.Bloom;
189 Pipeline.Data.BloomAmount = Camera.Effects.BloomAmount;
190
191 Pipeline.MainFrameBuffer = Camera.GetMainFrameBuffer();
192 Pipeline.GBuffer = Camera.GetGBuffer();
193 Pipeline.HDRFrameBuffer = Camera.GetHDRFrameBuffer();
194 Pipeline.BlurFrameBuffers0 = Camera.GetBlurFrameBuffers(0);
195 Pipeline.BlurFrameBuffers1 = Camera.GetBlurFrameBuffers(1);
196}

198void scl::renderer::EndPipeline()
199{
200 if (Pipeline.LightsStorage.IsDirectionalLight && Pipeline.LightsStorage.DirectionalLight.IsShadows) ComputeDepth(); // Shadow pass
201 ComputeGeometry(); // Geometry pass
202 ComputateLighting(); // Lighting pass
203
204 if (Pipeline.Data.IsHDR && Pipeline.Data.IsBloom) ComputeBloom(); // Bloom pass
205 if (Pipeline.Data.IsHDR) ComputeToneMapping(); // Tone mapping pass
206
207 Pipeline.Clear();
208}
